class Solution
{
public:
    int uniquePaths(int m, int n)
    {
        if (m <= 0 || n <= 0)
        {
            return 0;
        }
        if (m == 1 || n == 1)
        {
            return 1;
        }

        vector<vector<int>> dp(m, vector<int>(n)); //åˆ›å»ºå¤§å°ä¸ºm*nçš„æ•°ç»„
        // åˆå§‹åŒ–
        for (int i = 0; i < m; i++)
        {
            dp[i][0] = 1;
        }
        for (int i = 0; i < n; i++)
        {
            dp[0][i] = 1;
        }
        // æ¨å¯¼å‡º dp[m-1][n-1]
        for (int i = 1; i < m; i++)
        {
            for (int j = 1; j < n; j++)

            }
        }
        return dp[m - 1][n - 1];
    }
};

class Solution {
public:
    int uniquePaths(int m, int n) {
        if (m <= 0 || n <= 0)
        {
            return 0;
        }
        if (m == 1 || n == 1)
        {
            return 1;
        }

        vector<int> dp(n);
        // åˆå§‹åŒ–
        for (int i = 0; i < n; i++)
        {
            dp[i] = 1;
        }
        // æ¨å¯¼å‡º dp[m-1][n-1]
        for (int i = 1; i < m; i++)
        {
            // ç¬¬ i è¡Œç¬¬ 0 åˆ—çš„åˆå§‹å€¼
            dp[0] = 1;
            for (int j = 1; j < n; j++)
            {
                dp[j] = dp[j - 1] + dp[j];
            }
        }
        return dp[n - 1];
    }
};
